## Initializing popup

> index html

<pre><code>
    <div class="popup-container">     
        <a href="images_path_here">
            <img src="images_path_here" alt="" />
        </a>
    </div>
</code></pre>



> main js or active js
[`// Initialize MagnificPopup Plugin
    $('.popup-container').magnificPopup({
        delegate: 'a',
        type:'image',
        gallery:{enabled:true},
        zoom:{
            enabled:true,
            duration: 300,
            easing: 'ease-in-out'
        }
    });`]
